.586
.model flat, stdcall
includelib kernel32.lib
ExitProcess PROTO : DWORD
WriteConsoleA PROTO :DWORD, :DWORD, :DWORD, :DWORD, :DWORD
GetStdHandle PROTO :DWORD

STD_OUTPUT_HANDLE EQU -11  ; Дескриптор для стандартного вывода консоли

.stack 4096

.const
	__L0 SDWORD 12
	__L1 SDWORD 100
	__L2 SDWORD 13
	__L3 SDWORD 12
	__L4 SDWORD 90
	__L5 SDWORD 1000
	__L6 SDWORD 32

.data
buffer db "result: ", 10 DUP(0)  ; Буфер для сообщения с результатом, включая место для числа
bytesWritten DWORD 0             ; Переменная для записи количества байт
	a SDWORD 0
	f SDWORD 0
	x SDWORD 0


.code

main proc
start:
	mov eax, __L0
	mov [a], eax

	; original expression:  a+__L1
	mov eax, [a]
	add eax, __L1
	mov [f], eax

	; original expression:  a+f
	mov eax, [a]
	add eax, [f]
	mov [x], eax

	; original expression:  __L2+x-__L3-__L4+__L5-__L6
	mov eax, __L2
	add eax, [x]
	sub eax, __L3
	sub eax, __L4
	add eax, __L5
	sub eax, __L6
	mov [f], eax

	mov eax, [x]
	call ConvertToString            ; Преобразуем число в строку
	call PrintToConsole             ; Выводим строку в консоль

	mov eax, [f]
	call ConvertToString            ; Преобразуем число в строку
	call PrintToConsole             ; Выводим строку в консоль

	push 0
	call ExitProcess
main ENDP
end main
